import numpy as np
import matplotlib.pyplot as plt

# Generate random data for X, Y, and sizes
np.random.seed(42)  # For reproducibility
x = np.random.rand(100) * 100  # Random values for X (0-100)
y = np.random.rand(100) * 100  # Random values for Y (0-100)
sizes = np.random.rand(100) * 1000  # Random sizes for the balls (0-1000)

# Create scatter plot with varying ball sizes
plt.scatter(x, y, s=sizes, color='blue', alpha=0.6, edgecolor='k')

# Add labels and title
plt.xlabel('X Values')
plt.ylabel('Y Values')
plt.title('Scatter Plot with Balls of Different Sizes')

# Show the grid
plt.grid(True, linestyle='--', alpha=0.6)

# Show the plot
plt.tight_layout()
plt.show()
